--MOVIE:
CREATE SEQUENCE S1
START WITH 0
INCREMENT BY 1
MINVALUE 0
NOMAXVALUE
NOCYCLE;

CREATE TRIGGER MOVE_S1
BEFORE INSERT ON MOVIE
FOR EACH ROW
BEGIN
SELECT S1.NEXTVAL
INTO :NEW.ID
FROM DUAL;
END;
/
